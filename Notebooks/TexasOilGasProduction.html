List<InlineDatasetElement> dataset = new List<InlineDatasetElement>(ProductionData.Count);

foreach (WellProduction productionData in ProductionData)
{
    WellProductionRecord record;

    for(int i = 0; i < productionData.Records.Count; ++i)
    {
        record = productionData.Records[i];

        dataset.Add(new Dictionary<string, object>(4)
        {
            {
                "API", productionData.Api.ToString()
            },
            {
                "Month", record.Month
            },
            {
                "MonthlyBOE", record.MonthlyOil + (record.MonthlyGas / 5.8)
            }
        });
    }
}
